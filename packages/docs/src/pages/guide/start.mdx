### Ready to Go!

Now, you can create player to streaming video`:

<Tabs items={['vanilla', 'script']}>
  <Tab>
  ```js filename="app.js"
import Player from '@oplayer/core'
import ui from '@oplayer/ui'
import hls from '@oplayer/hls'

Player.make('#oplayer', {
source: {
src: 'https://oplayer.vercel.app/君の名は.mp4',
poster: 'https://oplayer.vercel.app/poster.png'
}
})
.use([ui(), hls()])
.create()

````
  </Tab>
  <Tab>
    ```html
  <script>
  OPlayer.make('#oplayer', {
    source: {
      src: 'https://oplayer.vercel.app/君の名は.mp4',
      poster: 'https://oplayer.vercel.app/poster.png'
    }
  })
    .use([OPlayer.ui(), OHls()])
    .create()
</script>
    ```
